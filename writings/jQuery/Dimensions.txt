# Dimensions

	$(...).css('width'); // returns '400px';
	
	$(...).width(); // returns 400;
	$(...).width(value); // set width
	
	$(...).innnerWidth(); // width + padding
	$(...).outerWidth(); // width + padding + border
	$(...).outerWidth(true); // width + padding + border + margin
	
	$(...).height();
	$(...).innerHeight();
	$(...).outerHeight(...);
